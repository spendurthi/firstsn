CREATE TABLE PUBLIC.DEPARTMENT (
	DEPT_ID BIGINT NOT NULL,
	DEPT_NAME VARCHAR(30),
	PRIMARY KEY (DEPT_ID)
);

CREATE TABLE PUBLIC.EMPLOYEE (
	EMP_ID INTEGER NOT NULL,
	EMP_NAME VARCHAR(25),
	DEPT_ID BIGINT,
	PRIMARY KEY (EMP_ID)
);

ALTER TABLE PUBLIC.EMPLOYEE
	ADD FOREIGN KEY (DEPT_ID) 
	REFERENCES DEPARTMENT (DEPT_ID);
	
CREATE TABLE PUBLIC.ADDRESS (
	ADD_ID INTEGER NOT NULL,
	STREET VARCHAR(25),
	CITY VARCHAR(16),
	STATE VARCHAR(16),
	ZIP VARCHAR(8),
	COUNTRY VARCHAR(20),
	PHONE_HOME VARCHAR(16),
	PHONE_OFFICE VARCHAR(16),
	PHONE_MOBILE VARCHAR(16),
	FAX VARCHAR(16),
	EXTENSION VARCHAR(8),
	ADD_TYPE VARCHAR(5),
	PRIMARY KEY (ADD_ID)
);

CREATE TABLE PUBLIC.DESIGNATIONS (
	DESIG_ID INTEGER NOT NULL,
	DESIG_LEVEL VARCHAR(25) NOT NULL,
	DESIG_NAME VARCHAR(25) NOT NULL,
	DEPT_ID BIGINT,
	PRIMARY KEY (DESIG_ID,DESIG_LEVEL,DESIG_NAME)
);

ALTER TABLE PUBLIC.DESIGNATIONS
	ADD FOREIGN KEY (DEPT_ID) 
	REFERENCES DEPARTMENT (DEPT_ID);

CREATE TABLE PUBLIC.ORGANIZATION (
	ORG_ID INTEGER NOT NULL,
	ORG_NAME VARCHAR(25),
	ORG_LOGO BLOB(16777216),
	ADD_ID INTEGER,
	PRIMARY KEY (ORG_ID)
);

ALTER TABLE PUBLIC.ORGANIZATION
	ADD FOREIGN KEY (ADD_ID) 
	REFERENCES ADDRESS (ADD_ID);
